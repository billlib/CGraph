106  CBool tryPop(T& value) {
107  // 这里不使用raii锁，主要是考虑到多线程的情况下，可能会重复进入
108  // raii锁指的是类似于lock_guard的用法，锁和对象的构造时获取、析构时释放
109  // 此处直接使用mutex_的try_lock和unlock，手动管理加锁和释放，因此非RAII锁

114  deque_.pop_front();

116  }
117  mutex_.unlock();
118  }
